This notebook shows that guessing the Hamming weight $h'$ progressively and guessing nonzero-positions+nonzero values is beneficial to just fixing h' and guessing the latter two.

We will proceed on a concrete example we'd like to break: kyber (256,2,25).

Consider $n=512$ dimensional lwe instance with Binomial(2) distribution and exactly $h=25$ nonzero coodinates in $\vec{s}$. We'd like to guess the nonzero coordinates within the last $k=109$ coordinates of $\vec{s}$. If we fail to do so, we'd need to permute the coordinates of $\vec{s}$ and vectors of $\mat(A)$ accordingly (this yields in a BKZ reduction).

Below, I provide the three following figures comparing the progressive vs. single-weight guessing:

  • Fig.1: success probability per a single permutation of the coordinates.
  • Fig.2: number of guesses (positions+nonzero values).
  • Fig.3: overhead of the number of guesses compared to the single-guess strategy.

def drop(n, h, k, fail=0, rotations=False):
    """
    Probability that ``k`` randomly sampled components have ``fail`` non-zero components amongst
    them.

    :param n: LWE dimension `n > 0`
    :param h: number of non-zero components
    :param k: number of components to ignore
    :param fail: we tolerate ``fail`` number of non-zero components amongst the `k` ignored
        components
    :param rotations: consider rotations of the basis to exploit ring structure (NTRU only)
    """

    N = n  # population size
    K = n - h  # number of success states in the population
    n = k  # number of draws
    k = n - fail  # number of observed successes
    prob_drop = binomial(K, k) * binomial(N - K, n - k) / binomial(N, n)
    if rotations:
        return 1 - (1 - prob_drop) ** N
    else:
        return prob_drop

def draws_for_confidence(n, k, w, h, confidence):
    p = binomial(n - k, w - h) * binomial(k, h) / binomial(n, w)
    return 1 if p > confidence else ceil(log(1 - confidence) / log(1 - p))


def required_iterations(params, success_probability=0.99):
    N, k, w, h_ = params["n"] * params.get("k_dim", 1), params['k'], params['w'], params['h_']
    return draws_for_confidence(N, k, w, h_, success_probability)

for w in range(9):
    proba = ( binomial(k,w) ) * binomial(n-k,h-w) / binomial(n,h) #the probability of last k coordinates being of hw exactly w    
    cumproba += proba.n()
    probas[w] = proba
    cumprobas[w] = cumproba
    
    curn = binomial(k,w) * 4**( 2*eta ) #the numbber of guesses
    N+=curn
    Ns[w] = curn
    cumNs[w] = N

    guess_overheads[w] = ((N-curn)/curn) #in percentage of curn
    
    # print(f"w:{w}, pr: {proba.n()}, cumpr:{cumproba}, num={curn}, ")
    req_it_single_weight = required_iterations({'n':n,'k_dim':1, 'k':k, 'w':h, 'h_':w })
    req_it_cum_weight = ceil(log(1 - confidence) / log(1 - cumproba))

    req_it_single_weights[w] = req_it_single_weight
    req_it_cum_weights[w] = req_it_cum_weight
